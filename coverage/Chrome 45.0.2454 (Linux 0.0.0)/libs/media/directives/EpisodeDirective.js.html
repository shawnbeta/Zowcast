'use strict';
 
hcMedia.directive('episodeCollection', function() {
 
    return {
        require: '?ngModel',
        scope: {},
        replace: true,
        controller: 'EpisodeController',
        controllerAs: 'episodeCtrl',
        bindToController: true,
        restrict : 'E',
 
        link: function(scope, element, attributes, ngModelCtrl){
            element.addClass('button');
 
            ngModelCtrl.$render = function() {
                element.toggleClass('primary', angular.equals(ngModelCtrl.$modelValue, scope.$eval(attributes.radioButton)));
            };
 
            element.bind('click', function() {
                if (!element.hasClass('active')) {
                    scope.$apply(function () {
                        ngModelCtrl.$setViewValue(scope.$eval(attributes.radioButton));
                        ngModelCtrl.$render();
                    });
                }
            });
        },
 
        templateUrl: 'src/templates/media/episodes/collection.html'
    };
});
 
 
hcMedia.directive('episodeItem', function() {
 
    return {
 
        scope: {},
        replace: true,
        controller: 'EpisodeController',
        controllerAs: 'episodeCtrl',
        bindToController: true,
        restrict : 'E',
 
        link: function(scope){
            scope.subscriptionType = 'audio';
            scope.options = [ 'audio', 'video', 'green' ];
        },
 
        templateUrl: 'src/templates/media/episodes/item.html'
    };
});

(function() {
    'use strict';
    angular.module('hcApp.media')
        .factory('EpisodeService', EpisodeService);
 
 
    function EpisodeService(HelperService) {
 
        return {
 
            helperService: this.helperService || HelperService,
 
            updateRemote: function (rqst) {
                var path = 'episodes/update';
                return helperService.makeRequest(path, rqst);
            },
 
            deleteEpisodeBySubscription: function (collection, subscriptionId) {
                var subEpisodes = [];
                _.each(collection, function (e, i) {
                    if (e.subscription_id == subscriptionId) {
                        subEpisodes.push(i)
                    }
                });
                return _.omit(collection, subEpisodes);
            },
 
            deleteOldEpisodes: function (collection, dateToCompare) {
                var oldItems = [];
                _.each(collection, function (e, i, d) {
                    if (e.pubDate < dateToCompare) {
                        oldItems.push(i)
                    }
                });
                return _.omit(collection, oldItems);
            }
 
        }
 
 
    };
});

hcApp.config(['$routeProvider',
 
    function($routeProvider) {
        $routeProvider
            .when('/', {
                controller : 'EmptyController',
                templateUrl : 'app/src/templates/episodes.html'
            })
 
            .when('/episodes', {
                controller : 'EmptyController',
                templateUrl : 'app/src/templates/episodes.html'
            })
 
            .when('/subscriptions', {
                controller : 'EmptyController',
                templateUrl : 'app/src/templates/subscriptions.html'
            })
 
            .when('/add', {
                controller : 'EmptyController',
                templateUrl : 'app/src/templates/add.html'
            })
 
            .when('/about', {
                controller : 'EmptyController',
                templateUrl : 'app/src/templates/about.html'
            })
 
            .when('/settings', {
                controller : 'EmptyController',
                templateUrl : 'app/src/templates/settings.html'
            })
 
            .otherwise({
                redirectTo : '/'
            });
    }]);

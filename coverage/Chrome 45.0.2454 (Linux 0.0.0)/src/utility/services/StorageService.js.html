hcUtilities.service('StorageService', [
 
    function(){
 
            this.saveString = function(id, obj){
                localStorage.setItem(id, JSON.stringify(obj));
            };
 
        this.loadObj = function(_id){
                var dataString = localStorage.getItem(_id);
                return JSON.parse(dataString);
            }
    }
]);
